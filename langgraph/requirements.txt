# LangGraph 学习示例依赖清单
# 安装: pip install -r requirements.txt (建议先建 venv)
# 版本以本地实测为准 (langgraph 1.2.7 / langchain 1.x), 用 >= 做下限约束, 便于后续小版本升级。

# --- 核心: 图运行时 + checkpointer ---
langgraph>=1.2.7,<2.0            # 图构建/运行时: StateGraph、Send、Command、interrupt、Pregel、func API
langgraph-checkpoint-sqlite>=3.1.0  # SqliteSaver: 落盘型 checkpointer, 崩溃恢复/持久化示例用

# --- LangChain 主框架 + 模型接入 ---

langchain-anthropic>=1.4.0,<2.0 # ChatAnthropic: 统一模型接入, model_id/base_url/key 全走 .env

# --- 配置 ---
python-dotenv>=1.2.0            # load_dotenv(override=True) 读取 .env

# --- 多 Agent 联网搜索工具 (multi_agent.py 用到) ---
langchain-community>=0.3.0      # DuckDuckGoSearchRun 工具封装
duckduckgo-search>=6.0.0        # DuckDuckGoSearchRun 的实际检索后端 (联网; 无网时该示例走替身)

# --- 数据校验 (结构化输出 schema) ---
pydantic>=2.13.0,<3.0           # BaseModel/Field 定义结构化输出 schema
